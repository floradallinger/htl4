[source, bash]
----
kubectl cluster-info
Kubectl get all
kubectl get nodes
kubectl get pods –all-namespaces
kubectl create namespace k8stest
kubectl apply -f <file>.yml
kubectl apply -f target/kubernetes/kubernetes.yml -n k8stest
kubectl logs [POD] -n [NAME_SPACE] --all-containers
----
